    <script>
        // 视频错误处理 (控制台提示，方便排查)
        document.getElementById('main-video').addEventListener('error', function() {
            console.warn("视频加载失败。请确保您已将视频下载到本地，重命名为 bg-video.mp4，并与此 HTML 文件放在同一文件夹中。");
        }, true);

        // --- 粒子特效逻辑 ---
        const canvas = document.getElementById('particle-canvas');
        const ctx = canvas.getContext('2d');

        function resizeCanvas() {
            canvas.width = window.innerWidth;
            canvas.height = window.innerHeight;
        }
        window.addEventListener('resize', resizeCanvas);
        resizeCanvas();

        let particlesArray = [];

        class Particle {
            constructor() {
                // 粒子集中在屏幕左侧 (配合登录框位置)
                this.x = Math.random() * (canvas.width * 0.45);
                this.y = canvas.height + Math.random() * 50; 
                this.size = Math.random() * 2.5 + 0.5;
                this.speedX = Math.random() * 1.5 - 0.75; 
                this.speedY = Math.random() * -2 - 1; 
                const colors = ['255, 69, 0', '255, 140, 0', '255, 215, 0', '255, 100, 50'];
                this.baseColor = colors[Math.floor(Math.random() * colors.length)];
                this.opacity = 1; 
                this.fadeSpeed = Math.random() * 0.01 + 0.005; 
            }
            update() {
                this.x += this.speedX;
                this.y += this.speedY;
                this.opacity -= this.fadeSpeed;
            }
            draw() {
                ctx.fillStyle = `rgba(${this.baseColor}, ${this.opacity})`;
                ctx.beginPath();
                ctx.arc(this.x, this.y, this.size, 0, Math.PI * 2);
                ctx.fill();
            }
        }

        function initParticles() {
            const particleCount = Math.floor(window.innerWidth / 8);
            for (let i = 0; i < particleCount; i++) {
                particlesArray.push(new Particle());
            }
        }

        function animateParticles() {
            ctx.clearRect(0, 0, canvas.width, canvas.height);
            for (let i = 0; i < particlesArray.length; i++) {
                particlesArray[i].update();
                particlesArray[i].draw();
                if (particlesArray[i].opacity <= 0 || particlesArray[i].y < 0) {
                    particlesArray.splice(i, 1);
                    particlesArray.push(new Particle());
                    i--; 
                }
            }
            requestAnimationFrame(animateParticles);
        }

        initParticles();
        animateParticles();
    </script>
